SELECT GL_JE_HEADERS.JE_HEADER_ID,
					GL_JE_HEADERS.LEDGER_ID,
					GL_JE_HEADERS.NAME,
					GL_JE_HEADERS.DESCRIPTION,
					GL_JE_HEADERS.STATUS,
					GL_JE_LINES.EFFECTIVE_DATE,
					gl_code_combinations.SEGMENT2,
					GL_JE_LINES.DESCRIPTION as DESCRIPTION_1,
					GL_JE_LINES.ACCOUNTED_DR,
					substr(GL_JE_HEADERS.DESCRIPTION,0,3) as DESCRIPTION_STORECODE,
					to_char(GL_JE_LINES.EFFECTIVE_DATE,'YYMMDD') as EFFECTIVE_DATE_CHAR
					FROM gl_code_combinations
					INNER JOIN GL_JE_LINES
					ON GL_JE_LINES.CODE_COMBINATION_ID = gl_code_combinations.CODE_COMBINATION_ID
					INNER JOIN GL_JE_HEADERS
					ON GL_JE_HEADERS.JE_HEADER_ID  = GL_JE_LINES.JE_HEADER_ID
					WHERE GL_JE_HEADERS.LEDGER_ID IN (2028, 2025)
					AND GL_JE_HEADERS.DESCRIPTION LIKE '%SA%'
					AND GL_JE_HEADERS.STATUS = 'P'
					AND GL_JE_LINES.EFFECTIVE_DATE BETWEEN to_date('01-Jul-2015')  AND to_date('30-Nov-2015') 
					and gl_code_combinations.SEGMENT2 in (
          '00862',
'00856',
'00860',
'00863',
'00859',
'00861',
'00857',
'00858',
'00855'
          )
					AND GL_JE_LINES.DESCRIPTION = 'CASH'
					ORDER BY GL_JE_HEADERS.LEDGER_ID,
					GL_JE_LINES.EFFECTIVE_DATE,
					gl_code_combinations.SEGMENT2

SELECT FND_FLEX_VALUES.FLEX_VALUE,
FND_FLEX_VALUES_tL.FLEX_VALUE_MEANING,
FND_FLEX_VALUES_tL.DESCRIPTION
FROM FND_FLEX_VALUES
INNER JOIN FND_FLEX_VALUES_tL
ON FND_FLEX_VALUES.FLEX_VALUE_ID = FND_FLEX_VALUES_tL.FLEX_VALUE_ID
WHERE FND_FLEX_VALUES.FLEX_VALUE_SET_ID = '1014924'
AND FND_FLEX_VALUES.FLEX_VALUE in (
'NESUP',
'NENEM',
'NESBSSJ',
'NESSA',
'NESBSCI',
'NESBSBA',
'NESBSPA',
'NEBSSCR',
'NESBSBL'          
)
