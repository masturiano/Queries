SELECT FND_FLEX_VALUES.FLEX_VALUE,
					FND_FLEX_VALUES_tL.FLEX_VALUE_MEANING,
					FND_FLEX_VALUES_tL.DESCRIPTION
					FROM FND_FLEX_VALUES
					INNER JOIN FND_FLEX_VALUES_tL
					ON FND_FLEX_VALUES.FLEX_VALUE_ID = FND_FLEX_VALUES_tL.FLEX_VALUE_ID
					WHERE FND_FLEX_VALUES.FLEX_VALUE_SET_ID = '1014924'
					AND FND_FLEX_VALUES.FLEX_VALUE = 'NENEM'
