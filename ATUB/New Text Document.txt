SELECT CUST_ACCT.CUST_ACCOUNT_ID,
  CUST.PARTY_NAME,
  CUST_ACCT.ACCOUNT_NUMBER,
  CUST_USES.LOCATION,
  CPCLASS.NAME,
  CUST_SITE.ORG_ID,
  CUST.JGZZ_FISCAL_CODE,
  CUST.ADDRESS1,CUST.ADDRESS2,CUST.ADDRESS3,CUST.ADDRESS4,CUST.CITY
FROM HZ_PARTIES CUST,
  HZ_CUST_ACCOUNTS CUST_ACCT,
  HZ_CUST_ACCT_SITES_ALL CUST_SITE,
  HZ_CUST_SITE_USES_ALL CUST_USES,
  HZ_LOCATIONS CUST_LOC,
  hz_customer_profiles CPROF,
  hz_cust_profile_classes CPCLASS
WHERE CUST_ACCT.CUST_ACCOUNT_ID = CUST_SITE.CUST_ACCOUNT_ID
AND CUST_SITE.CUST_ACCT_SITE_ID = CUST_USES.CUST_ACCT_SITE_ID
AND CUST_USES.CUST_ACCT_SITE_ID = CUST_LOC.LOCATION_ID(+)
AND CUST.PARTY_ID               = CUST_ACCT.PARTY_ID
AND CUST_ACCT.CUST_ACCOUNT_ID   = CPROF.CUST_ACCOUNT_ID
AND CPROF.PROFILE_CLASS_ID      = CPCLASS.PROFILE_CLASS_ID
AND CUST_USES.SITE_USE_ID       = CPROF.SITE_USE_ID
AND (CUST_SITE.ORG_ID          IN (85)
AND CUST_ACCT.ACCOUNT_NUMBER   IN (
'267742'
))
GROUP BY CUST_ACCT.CUST_ACCOUNT_ID,
  CUST.PARTY_NAME,
  CUST_ACCT.ACCOUNT_NUMBER,
  CUST_USES.LOCATION,
  CPCLASS.NAME,
  CUST_SITE.ORG_ID,
  CUST.JGZZ_FISCAL_CODE,
  CUST.ADDRESS1,CUST.ADDRESS2,CUST.ADDRESS3,CUST.ADDRESS4,CUST.CITY
ORDER BY CPCLASS.NAME,CUST_USES.LOCATION

--hz_cust_site_uses_all.location
select ar_payment_schedules_all.customer_site_use_id,hz_cust_acct_sites_all.cust_acct_site_id  from HZ_CUST_ACCOUNTS
left join HZ_CUST_ACCT_SITES_ALL on HZ_CUST_ACCT_SITES_ALL.CUST_ACCT_SITE_ID = HZ_CUST_ACCOUNTS.CUST_ACCOUNT_ID
left join HZ_CUST_SITE_USES_ALL on HZ_CUST_SITE_USES_ALL.CUST_ACCT_SITE_ID = HZ_CUST_ACCT_SITES_ALL.CUST_ACCT_SITE_ID
left join ar_payment_schedules_all on ar_payment_schedules_all.customer_site_use_id = hz_cust_acct_sites_all.cust_acct_site_id
left join ra_customer_trx_lines_all on ra_customer_trx_lines_all.CUSTOMER_TRX_ID = ar_payment_schedules_all
where HZ_CUST_ACCOUNTS.ACCOUNT_NUMBER   IN (
'267742'
)

desc ra_customer_trx_lines_all
desc RA_CUSTOMER_TRX_ALL
